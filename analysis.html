map_pts <- function(df, x = "lon_degrees", y = "lat_degrees", color = "ocean", legend.rows = 1){
  ggplot() +
  borders("world", fill = "gray80", colour = "gray80") +
  geom_point(data = df,
    aes(x = .data[[x]],
        y = .data[[y]],
        color = .data[[color]]), size = .5) +
  coord_sf() +
  theme_minimal() +
  theme(legend.position = "bottom") +
  guides(colour = guide_legend(ncol = legend.rows))
}

map_pts(sample_locs, color = "ocean", legend.rows = 3)
